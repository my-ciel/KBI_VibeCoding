javascript// 점수 계산 로직
class ScoreManager {
  calculateScore(isCorrect, timeSpent, consecutiveCorrect, hintUsed) {
    let score = 0;
    if (isCorrect) {
      score += 10; // 기본 점수
      if (timeSpent < 10) score += 3; // 시간 보너스
      if (!hintUsed) score += 2; // 노힌트 보너스
      score += this.getConsecutiveBonus(consecutiveCorrect);
    }
    return score;
  }
}
